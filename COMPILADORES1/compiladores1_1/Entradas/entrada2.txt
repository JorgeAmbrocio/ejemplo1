int soyFuncion (int contador) {
	
	if (contador != 1) {
		
        return contador * soyFuncion(contador - 1 ) ;
	}else if (contador == 1) {
            // sí es igual que uno
        
		return 1 ;
	}
}

int hola  = soyFuncion(5);

println("El valor es: " + hola);
